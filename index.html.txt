/* ========== REVEAL ========== */
function initReveal(){
  var obs = new IntersectionObserver(function(entries){
    for(var i = 0; i < entries.length; i++){
      if(entries[i].isIntersecting){
        (function(el, idx){ setTimeout(function(){ el.classList.add('visible'); }, idx * 80); })(entries[i].target, i);
        obs.unobserve(entries[i].target);
      }
    }
  }, {threshold: 0.15, rootMargin: '0px 0px -30px 0px'});
  var items = document.querySelectorAll('.reveal, .cred-item');
  for(var j = 0; j < items.length; j++) obs.observe(items[j]);
}

/* ========== RIPPLE ========== */
function addRipple(e){
  var btn = e.currentTarget, rect = btn.getBoundingClientRect();
  var size = Math.max(rect.width, rect.height) * 1.2;
  var cx = e.clientX || (e.touches && e.touches[0] ? e.touches[0].clientX : rect.left + rect.width/2);
  var cy = e.clientY || (e.touches && e.touches[0] ? e.touches[0].clientY : rect.top + rect.height/2);
  var rip = document.createElement('span');
  rip.className = 'ripple';
  rip.style.width = size + 'px'; rip.style.height = size + 'px';
  rip.style.left = (cx - rect.left - size/2) + 'px';
  rip.style.top = (cy - rect.top - size/2) + 'px';
  btn.appendChild(rip);
  setTimeout(function(){ rip.remove(); }, 600);
}
